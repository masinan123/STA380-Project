  if (alternative == "greater") {
    (sum(perm_stats >= observed) + 1) / (length(perm_stats) + 1)
    
  } else if (alternative == "less") {
    (sum(perm_stats <= observed) + 1) / (length(perm_stats) + 1)

    (sum(abs(perm_stats) >= abs(observed)) + 1) / (length(perm_stats) + 1)

  }
}
